  #코드 공유
  import re
  import sys
  
def input():
    return sys.stdin.readline().rstrip()

count = int(input())
stack = []
result = []

for i in range(count):
    s = input()
    if re.match('push.', s):
        num = re.sub('push ', '', s)
        stack.append(num)
        
    elif s in 'pop':
        if len(stack) != 0:
            result.append(stack.pop())
        else:
            result.append(-1)
            
    elif s in 'size':
        result.append(len(stack))
    
    elif s in 'empty':
        if len(stack) == 0:
            result.append(1)
        else:
            result.append(0)
    
    elif s in 'top':
        if len(stack) != 0:
            result.append(stack[-1])
        else:
            result.append(-1)  
        
for j in result:
    print(j)
