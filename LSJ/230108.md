  ```python
  #코드 공유
  import sys

def input():
    return sys.stdin.readline().rstrip()

n = int(input())
que = []

for i in range(n):
    s = input()
    
    if(s == "pop" or s == "front"):
        if (que):
            print(que[0])
            if(s == "pop"):
                del que[0]
        else:
            print(-1)
    
    elif(s == "size"):
        print(len(que))
    
    elif(s == "empty"):
        if(que):
            print(0)
        else:
            print(1)
            
    elif(s == "back"):
        if(que):
            print(que[len(que)-1])
        else:
            print(-1)
            
    else: # push x 의 경우 문자 일치를 판단하려면 정규표현식 사용해야해서 else로 처리함
        que.append(s.split(' ')[1])
    


  ```
